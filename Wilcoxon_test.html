# ======================================================
# Enhanced Model Comparison: ARIMA(1,1,0) vs mHC
# Using Wilcoxon Signed-Rank Test with Multiple Metrics
# ======================================================

import numpy as np
import pandas as pd
from scipy.stats import wilcoxon
from scipy.stats import shapiro, kstest
import matplotlib.pyplot as plt
from sklearn.metrics import mean_absolute_error, mean_squared_error
import warnings
warnings.filterwarnings('ignore')

# ======================================================
# Actual forecast values from both models
# ======================================================

# mHC forecast values (non-linear, volatile)
mhc_forecasts = np.array([23899.77, 25354.69, 24682.36, 25607.10, 27275.69,
                          25370.59, 25470.47, 25402.24, 23665.60, 25424.21])

# ARIMA(1,1,0) forecast values (linear, convergent)
arima_forecasts = np.array([23017.39, 23105.11, 23139.89, 23153.68, 23159.15,
                            23161.32, 23162.18, 23162.52, 23162.66, 23162.71])

# Load historical data
df = pd.read_excel(r"G:\2018\Documents\papers\New Methodologies\mHC\mHCdata.xlsx")
historical_values = df["GDP"].values[-10:]  # Last 10 historical values

print("=" * 80)
print("ENHANCED MODEL COMPARISON: ARIMA(1,1,0) vs mHC")
print("=" * 80)

# ======================================================
# METRIC 1: Forecast Volatility (Standard Deviation)
# ======================================================

arima_volatility = np.std(arima_forecasts)
mhc_volatility = np.std(mhc_forecasts)

# Create paired volatility measurements using rolling windows
volatility_pairs = []
for i in range(3, 11):  # Use different window sizes
    arima_roll_std = np.std(arima_forecasts[:i])
    mhc_roll_std = np.std(mhc_forecasts[:i])
    volatility_pairs.append((arima_roll_std, mhc_roll_std))

volatility_diffs = [mhc - arima for arima, mhc in volatility_pairs]

# ======================================================
# METRIC 2: Non-linearity Measure (Deviation from Linear Trend)
# ======================================================

# Fit linear trend to ARIMA forecasts (should be almost perfect)
arima_trend = np.polyfit(range(10), arima_forecasts, 1)
arima_linear_fit = np.polyval(arima_trend, range(10))
arima_nonlinearity = np.abs(arima_forecasts - arima_linear_fit)

# Fit linear trend to mHC forecasts (should show significant deviations)
mhc_trend = np.polyfit(range(10), mhc_forecasts, 1)
mhc_linear_fit = np.polyval(mhc_trend, range(10))
mhc_nonlinearity = np.abs(mhc_forecasts - mhc_linear_fit)

# ======================================================
# METRIC 3: Forecast Stability (Step-to-Step Changes)
# ======================================================

arima_changes = np.abs(np.diff(arima_forecasts))
mhc_changes = np.abs(np.diff(mhc_forecasts))

# ======================================================
# METRIC 4: Distribution Characteristics (using bootstrapping)
# ======================================================

np.random.seed(42)
n_bootstrap = 1000
arima_bootstrap_means = []
mhc_bootstrap_means = []

for _ in range(n_bootstrap):
    bootstrap_idx = np.random.choice(10, 10, replace=True)
    arima_bootstrap_means.append(np.mean(arima_forecasts[bootstrap_idx]))
    mhc_bootstrap_means.append(np.mean(mhc_forecasts[bootstrap_idx]))

# ======================================================
# Perform Wilcoxon Tests on Multiple Metrics
# ======================================================

print("\n" + "=" * 80)
print("WILCOXON SIGNED-RANK TESTS ON MULTIPLE METRICS")
print("=" * 80)

results = []

# Test 1: Non-linearity Measure
wilcoxon_stat_nl, wilcoxon_p_nl = wilcoxon(arima_nonlinearity, mhc_nonlinearity)
results.append({
    'Metric': 'Non-linearity',
    'ARIMA Mean': f"{np.mean(arima_nonlinearity):.2f}",
    'mHC Mean': f"{np.mean(mhc_nonlinearity):.2f}",
    'Statistic': f"{wilcoxon_stat_nl:.4f}",
    'P-value': f"{wilcoxon_p_nl:.6f}",
    'Significant': wilcoxon_p_nl < 0.05
})

# Test 2: Forecast Stability (Step Changes)
wilcoxon_stat_sc, wilcoxon_p_sc = wilcoxon(arima_changes, mhc_changes)
results.append({
    'Metric': 'Step Changes',
    'ARIMA Mean': f"{np.mean(arima_changes):.2f}",
    'mHC Mean': f"{np.mean(mhc_changes):.2f}",
    'Statistic': f"{wilcoxon_stat_sc:.4f}",
    'P-value': f"{wilcoxon_p_sc:.6f}",
    'Significant': wilcoxon_p_sc < 0.05
})

# Test 3: Bootstrap Means Distribution
# Take a subset of bootstrap samples for paired test
bootstrap_sample_size = 100
wilcoxon_stat_bs, wilcoxon_p_bs = wilcoxon(
    arima_bootstrap_means[:bootstrap_sample_size], 
    mhc_bootstrap_means[:bootstrap_sample_size]
)
results.append({
    'Metric': 'Bootstrap Means',
    'ARIMA Mean': f"{np.mean(arima_bootstrap_means):.2f}",
    'mHC Mean': f"{np.mean(mhc_bootstrap_means):.2f}",
    'Statistic': f"{wilcoxon_stat_bs:.4f}",
    'P-value': f"{wilcoxon_p_bs:.6f}",
    'Significant': wilcoxon_p_bs < 0.05
})

# Test 4: Volatility (using rolling windows)
wilcoxon_stat_vol, wilcoxon_p_vol = wilcoxon([p[0] for p in volatility_pairs], 
                                             [p[1] for p in volatility_pairs])
results.append({
    'Metric': 'Volatility',
    'ARIMA Mean': f"{np.mean([p[0] for p in volatility_pairs]):.2f}",
    'mHC Mean': f"{np.mean([p[1] for p in volatility_pairs]):.2f}",
    'Statistic': f"{wilcoxon_stat_vol:.4f}",
    'P-value': f"{wilcoxon_p_vol:.6f}",
    'Significant': wilcoxon_p_vol < 0.05
})

# Display results
results_df = pd.DataFrame(results)
print("\n", results_df.to_string(index=False))

# ======================================================
# Comprehensive Statistical Analysis
# ======================================================

print("\n" + "=" * 80)
print("COMPREHENSIVE STATISTICAL ANALYSIS")
print("=" * 80)

# Test for normality of original APE differences (to show why Wilcoxon is appropriate)
arima_ape = np.abs((arima_forecasts - historical_values) / historical_values) * 100
mhc_ape = np.abs((mhc_forecasts - historical_values) / historical_values) * 100
ape_differences = arima_ape - mhc_ape

shapiro_stat, shapiro_p = shapiro(ape_differences)
ks_stat, ks_p = kstest(ape_differences, 'norm')

print(f"\nNormality Tests for APE Differences:")
print(f"Shapiro-Wilk Test: p = {shapiro_p:.6f} {'(Non-normal)' if shapiro_p < 0.05 else '(Normal)'}")
print(f"Kolmogorov-Smirnov Test: p = {ks_p:.6f} {'(Non-normal)' if ks_p < 0.05 else '(Normal)'}")

# ======================================================
# Visual Demonstration of Model Differences
# ======================================================

fig, axes = plt.subplots(2, 3, figsize=(16, 12))

# Plot 1: Forecast Comparison with Confidence Bands
ax1 = axes[0, 0]
ax1.plot(range(1, 11), arima_forecasts, 'bs-', label='ARIMA(1,1,0)', linewidth=2, markersize=8)
ax1.plot(range(1, 11), mhc_forecasts, 'r^-', label='mHC', linewidth=2, markersize=8)
ax1.fill_between(range(1, 11), 
                 mhc_forecasts - np.std(mhc_forecasts), 
                 mhc_forecasts + np.std(mhc_forecasts), 
                 alpha=0.2, color='red', label='mHC 1')
ax1.fill_between(range(1, 11), 
                 arima_forecasts - np.std(arima_forecasts), 
                 arima_forecasts + np.std(arima_forecasts), 
                 alpha=0.2, color='blue', label='ARIMA 1')
ax1.set_xlabel('Step')
ax1.set_ylabel('GDP (US$ million)')
ax1.set_title('Forecast Comparison with Volatility Bands')
ax1.legend()
ax1.grid(True, alpha=0.3)

# Plot 2: Non-linearity Measure
ax2 = axes[0, 1]
bar_width = 0.35
x_pos = np.arange(10)
ax2.bar(x_pos - bar_width/2, arima_nonlinearity, bar_width, label='ARIMA', alpha=0.8, color='blue')
ax2.bar(x_pos + bar_width/2, mhc_nonlinearity, bar_width, label='mHC', alpha=0.8, color='red')
ax2.set_xlabel('Step')
ax2.set_ylabel('Deviation from Linear Trend')
ax2.set_title('Non-linearity Measure')
ax2.set_xticks(x_pos)
ax2.set_xticklabels(range(1, 11))
ax2.legend()
ax2.grid(True, alpha=0.3)

# Plot 3: Step Changes Distribution
ax3 = axes[0, 2]
step_change_data = [arima_changes, mhc_changes]
bp = ax3.boxplot(step_change_data, labels=['ARIMA', 'mHC'], patch_artist=True)
bp['boxes'][0].set_facecolor('lightblue')
bp['boxes'][1].set_facecolor('lightcoral')
ax3.set_ylabel('Absolute Step Change')
ax3.set_title('Forecast Stability (Step Changes)')
ax3.grid(True, alpha=0.3)

# Plot 4: Bootstrap Distributions
ax4 = axes[1, 0]
ax4.hist(arima_bootstrap_means, bins=30, alpha=0.5, label='ARIMA', color='blue', density=True)
ax4.hist(mhc_bootstrap_means, bins=30, alpha=0.5, label='mHC', color='red', density=True)
ax4.set_xlabel('Mean GDP')
ax4.set_ylabel('Density')
ax4.set_title('Bootstrap Distribution of Means')
ax4.legend()
ax4.grid(True, alpha=0.3)

# Plot 5: Cumulative Volatility
ax5 = axes[1, 1]
for i in range(3, 11):
    ax5.scatter([i], [np.std(arima_forecasts[:i])], color='blue', s=50, alpha=0.5)
    ax5.scatter([i], [np.std(mhc_forecasts[:i])], color='red', s=50, alpha=0.5)
ax5.plot(range(3, 11), [np.std(arima_forecasts[:i]) for i in range(3, 11)], 
         'b-', label='ARIMA', alpha=0.5)
ax5.plot(range(3, 11), [np.std(mhc_forecasts[:i]) for i in range(3, 11)], 
         'r-', label='mHC', alpha=0.5)
ax5.set_xlabel('Number of Steps')
ax5.set_ylabel('Cumulative Volatility')
ax5.set_title('Volatility Evolution')
ax5.legend()
ax5.grid(True, alpha=0.3)

# Plot 6: Q-Q Plot of Bootstrap Differences
ax6 = axes[1, 2]
bootstrap_diffs = np.array(mhc_bootstrap_means[:100]) - np.array(arima_bootstrap_means[:100])
stats.probplot(bootstrap_diffs, dist="norm", plot=ax6)
ax6.set_title('Q-Q Plot of Bootstrap Differences')

plt.tight_layout()

print("\n" + "=" * 80)
print("FINAL INTERPRETATION")

significant_metrics = sum([r['Significant'] for r in results])

print(f"""

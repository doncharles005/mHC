class ManifoldEmbedding(nn.Module):
    def __init__(self, latent_dim=2, output_dim=1):
        super().__init__()

        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, 8),
            nn.Tanh(),
            nn.Linear(8, 16),
            nn.Tanh(),
            nn.Linear(16, output_dim)
        )

        self.encoder = nn.Sequential(
            nn.Linear(output_dim, 16),
            nn.Tanh(),
            nn.Linear(16, 8),
            nn.Tanh(),
            nn.Linear(8, latent_dim)
        )

    def forward(self, z):
        return self.decoder(z)

    def encode(self, h):
        return self.encoder(h)

    def project(self, h):
        return self.forward(self.encode(h))

# ======================================================
# Hyper-Connection
# ======================================================

class HyperConnection(nn.Module):
    def __init__(self, input_dim, hidden_dim=8):
        super().__init__()

        self.hyper_net = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.Tanh(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.Tanh(),
            nn.Linear(hidden_dim, 1)
        )

    def forward(self, current_z, previous_zs):

        if len(previous_zs) == 0:
            return current_z

        weights = []
        for z_prev in previous_zs:
            combined = torch.cat([current_z, z_prev], dim=1)
            weights.append(self.hyper_net(combined))

        weights = torch.softmax(torch.cat(weights, dim=1), dim=1)

        weighted_sum = torch.zeros_like(current_z)
        for i, z_prev in enumerate(previous_zs):
            weighted_sum += weights[:, i:i+1] * z_prev

        return current_z + 0.4 * weighted_sum  #  stronger influence

# ======================================================
# mHC Layer (WITH DYNAMICS + SHOCKS)
# ======================================================

class mHCLayer(nn.Module):
    def __init__(self, latent_dim=2):
        super().__init__()

        self.manifold = ManifoldEmbedding(latent_dim, output_dim=1)
        self.hyper = HyperConnection(input_dim=latent_dim * 2)

        self.transition = nn.Sequential(
            nn.Linear(latent_dim, latent_dim),
            nn.Tanh()
        )

    def forward(self, h, previous_zs):

        z = self.manifold.encode(h)
        z_tilde = self.hyper(z, previous_zs)

        z_next = 0.8 * z_tilde + 0.2 * self.transition(z_tilde)

        if not self.training:
            z_next = z_next + 0.05 * torch.randn_like(z_next)

        h_next = self.manifold(z_next)

        return h_next, z_next

# mHC Forecaster
# ======================================================

class mHCForecaster(nn.Module):
    def __init__(self, n_lags=5, latent_dim=2, n_layers=3):
        super().__init__()

        self.input_encoder = nn.Linear(n_lags, latent_dim)

        self.layers = nn.ModuleList([
            mHCLayer(latent_dim) for _ in range(n_layers)
        ])

        self.output_decoder = nn.Linear(latent_dim, 1)

    def forward(self, x, return_all=False):

        #  CRITICAL FIX: anchor to last observed value
        h = x[:, -1:].clone()
        z = self.input_encoder(x)

        previous_zs = []
        all_hs, all_zs, manifolds = [h], [z], []

        for layer in self.layers:
            h, z = layer(h, previous_zs)
            previous_zs.append(z)
            all_hs.append(h)
            all_zs.append(z)
            manifolds.append(layer.manifold)

        y_hat = self.output_decoder(z)

        if return_all:
            return y_hat, all_hs, all_zs, manifolds

        return y_hat

# ======================================================
# Manifold Regularization
# ======================================================

def manifold_regularization_loss(hs, zs, manifolds):
    loss = 0
    for h, z, m in zip(hs, zs, manifolds):
        recon = F.mse_loss(h, m.project(h))
        z_eps = z + 0.01 * torch.randn_like(z)
        smooth = F.mse_loss(m(z), m(z_eps))
        loss += recon + 0.1 * smooth
    return loss / len(hs)

def train_model(model, X, y, epochs=1000, lr=0.005):

    opt = torch.optim.Adam(model.parameters(), lr=lr)
    scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(opt, patience=50)

    losses = []

    for e in range(epochs):
        model.train()
        opt.zero_grad()

        pred, hs, zs, mfs = model(X, return_all=True)
        task_loss = F.mse_loss(pred, y)
        manifold_loss = manifold_regularization_loss(hs, zs, mfs)

        loss = task_loss + 0.03 * manifold_loss  #  weaker smoothing
        loss.backward()
        opt.step()
        scheduler.step(loss)

        losses.append(loss.item())

        if e % 100 == 0:
            print(f"Epoch {e} | Loss {loss.item():.6f}")

    return losses

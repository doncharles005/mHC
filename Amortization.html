# Calculate annual payment using the loan payment formula
# PMT = P * (r(1+r)^n) / ((1+r)^n - 1)
monthly_interest_rate = annual_interest_rate / payments_per_year
total_payments = loan_term_years * payments_per_year

# Annual payment calculation
if monthly_interest_rate == 0:
    annual_payment = principal / total_payments
else:
    annual_payment = (principal * monthly_interest_rate * 
                     (1 + monthly_interest_rate) ** total_payments) / \
                     ((1 + monthly_interest_rate) ** total_payments - 1)

print(f"\nAnnual Payment: ${annual_payment:,.6f} million")
print(f"Total Payments over {loan_term_years} years: ${annual_payment * total_payments:,.6f} million")
print(f"Total Interest Paid: ${(annual_payment * total_payments) - principal:,.6f} million")

# CREATE AMORTIZATION SCHEDULE
# ======================================================

def create_amortization_schedule(principal, annual_rate, years, payment_freq=1):
    """Create a detailed amortization schedule"""
    
    rate_per_period = annual_rate / payment_freq
    total_periods = years * payment_freq
    
    # Calculate periodic payment
    if rate_per_period == 0:
        periodic_payment = principal / total_periods
    else:
        periodic_payment = (principal * rate_per_period * 
                           (1 + rate_per_period) ** total_periods) / \
                           ((1 + rate_per_period) ** total_periods - 1)
    
    # Initialize schedule
    schedule = []
    remaining_balance = principal
    
    for period in range(1, total_periods + 1):
        # Calculate interest for this period
        interest_payment = remaining_balance * rate_per_period
        
        # Calculate principal payment for this period
        principal_payment = periodic_payment - interest_payment
        
        # Ensure principal payment doesn't exceed remaining balance
        if principal_payment > remaining_balance:
            principal_payment = remaining_balance
            periodic_payment = principal_payment + interest_payment
        
        # Update remaining balance
        remaining_balance -= principal_payment
        
        # Ensure remaining balance doesn't go below zero
        if remaining_balance < 0:
            remaining_balance = 0
            principal_payment = periodic_payment - interest_payment
        
        # Calculate cumulative totals
        cumulative_principal = principal - remaining_balance
        cumulative_interest = (periodic_payment * period) - cumulative_principal
        
        # Add to schedule
        schedule.append({
            'Period': period,
            'Year': int(np.ceil(period / payment_freq)),
            'Payment_Number': period,
            'Beginning_Balance': remaining_balance + principal_payment,
            'Payment': periodic_payment,
            'Principal': principal_payment,
            'Interest': interest_payment,
            'Ending_Balance': remaining_balance,
            'Cumulative_Principal': cumulative_principal,
            'Cumulative_Interest': cumulative_interest,
            'Principal_Percentage': (principal_payment / periodic_payment) * 100,
            'Interest_Percentage': (interest_payment / periodic_payment) * 100
        })
    
    return pd.DataFrame(schedule), periodic_payment

# Create amortization schedule
amort_schedule, periodic_payment = create_amortization_schedule(

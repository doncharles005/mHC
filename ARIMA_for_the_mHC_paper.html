#import the libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.tsa.stattools import adfuller
from statsmodels.graphics.tsaplots import plot_acf, plot_pacf
import warnings
warnings.filterwarnings('ignore')

# Load the data
df = pd.read_excel(r"G:\2018\Documents\papers\New Methodologies\mHC\mHCdata.xlsx")

# Display basic information about the data
print("Data Shape:", df.shape)
print("\nFirst few rows:")
print(df.head())
print("\nData Info:")
print(df.info())

# Check for missing values in GDP column
print("\nMissing values in GDP:", df["GDP"].isnull().sum())

# Plot the GDP time series
plt.figure(figsize=(12, 6))
plt.plot(df["GDP"], marker='o', linestyle='-', linewidth=1, markersize=3)
plt.title('GDP Time Series')
plt.xlabel('Time Period')
plt.ylabel('GDP')
plt.grid(True, alpha=0.3)
plt.show()

# Check stationarity with Augmented Dickey-Fuller test
def check_stationarity(timeseries, title):
    # Perform ADF test
    result = adfuller(timeseries.dropna())
    print(f'\n{title} - Augmented Dickey-Fuller Test:')
    print(f'ADF Statistic: {result[0]:.6f}')
    print(f'p-value: {result[1]:.6f}')
    print(f'Critical Values:')
    for key, value in result[4].items():
        print(f'\t{key}: {value:.3f}')
    
    if result[1] <= 0.05:
        print("Conclusion: Series is stationary (reject H0)")
    else:
        print("Conclusion: Series is non-stationary (fail to reject H0)")
    
    return result[1] <= 0.05

# Check stationarity of original series
is_stationary = check_stationarity(df["GDP"], "Original GDP")

# If series is not stationary, apply differencing
if not is_stationary:
    # First order differencing
    df["GDP_diff"] = df["GDP"].diff()
    
    # Check stationarity of differenced series
    is_stationary_diff = check_stationarity(df["GDP_diff"], "Differenced GDP")
    
    # Plot original vs differenced series
    fig, axes = plt.subplots(2, 1, figsize=(12, 8))
    
    axes[0].plot(df["GDP"], color='blue')
    axes[0].set_title('Original GDP Series')
    axes[0].set_ylabel('GDP')
    axes[0].grid(True, alpha=0.3)
    
    axes[1].plot(df["GDP_diff"], color='red')
    axes[1].set_title('First Difference of GDP')
    axes[1].set_ylabel('Differenced GDP')
    axes[1].grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.show()
    
    # Use differenced series for ACF/PACF analysis
    series_for_acf = df["GDP_diff"].dropna()
    d = 1
else:
    series_for_acf = df["GDP"].dropna()
    d = 0

# Plot ACF and PACF to determine p and q
fig, axes = plt.subplots(1, 2, figsize=(15, 4))

# ACF plot
plot_acf(series_for_acf, ax=axes[0], lags=min(20, len(series_for_acf)//2))
axes[0].set_title('Autocorrelation Function (ACF)')

# PACF plot
plot_pacf(series_for_acf, ax=axes[1], lags=min(20, len(series_for_acf)//2))
axes[1].set_title('Partial Autocorrelation Function (PACF)')

plt.tight_layout()
plt.show()

# Based on ACF/PACF, you can try different ARIMA models
# Here's a systematic approach to find the best ARIMA model

# Split data into train and test (80-20 split for evaluation)
train_size = int(len(df) * 0.8)
train, test = df["GDP"][:train_size], df["GDP"][train_size:]

print(f"\nTrain size: {len(train)}, Test size: {len(test)}")

# Try different ARIMA models and select based on AIC
best_aic = np.inf
best_order = None
best_model = None

# Define ranges for p, d, q
p_range = range(0, 4)
d_range = [d]  # Use the d determined from stationarity test
q_range = range(0, 4)

print("\nSearching for best ARIMA model...")
for p in p_range:
    for d in d_range:
        for q in q_range:
            try:
                model = ARIMA(train, order=(p, d, q))
                fitted_model = model.fit()
                if fitted_model.aic < best_aic:
                    best_aic = fitted_model.aic
                    best_order = (p, d, q)
                    best_model = fitted_model
                print(f'ARIMA{ (p, d, q)} - AIC: {fitted_model.aic:.2f}')
            except:
                continue

print(f"\nBest ARIMA model: ARIMA{best_order} with AIC: {best_aic:.2f}")

# Fit the best model on the entire dataset
final_model = ARIMA(df["GDP"], order=best_order)
final_results = final_model.fit()

print("\nBest Model Summary:")
print(final_results.summary())

# Generate forecasts
# For in-sample predictions
df["Fitted"] = final_results.fittedvalues

# For out-of-sample forecasts (10 steps ahead)
n_forecast = 10
forecast = final_results.forecast(steps=n_forecast)
forecast_index = range(len(df), len(df) + n_forecast)

# Plot results
plt.figure(figsize=(14, 7))
plt.plot(df.index, df["GDP"], label='Actual GDP', color='blue', marker='o', markersize=3)
plt.plot(df.index, df["Fitted"], label='Fitted Values', color='green', linestyle='--')
plt.plot(forecast_index, forecast, label='Forecast', color='red', marker='s', markersize=4)
plt.axvline(x=len(df)-1, color='black', linestyle=':', alpha=0.7, label='Forecast Start')
plt.title(f'ARIMA{best_order} Model - GDP Forecast')
plt.xlabel('Time Period')
plt.ylabel('GDP')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()

# Display forecast results
print(f"\n{'-'*50}")
print(f"ARIMA{best_order} - 10-Step Ahead Forecast:")
print(f"{'-'*50}")
for i, (idx, value) in enumerate(zip(forecast_index, forecast)):
    print(f"Step {i+1} (Period {idx}): {value:.2f}")

# Calculate forecast confidence intervals
forecast_result = final_results.get_forecast(steps=n_forecast)
forecast_ci = forecast_result.conf_int()

# Plot with confidence intervals
plt.figure(figsize=(14, 7))
plt.plot(df.index, df["GDP"], label='Historical GDP', color='blue')
plt.plot(forecast_index, forecast, label='Forecast', color='red', marker='s')
plt.fill_between(forecast_index, 
                 forecast_ci.iloc[:, 0], 
                 forecast_ci.iloc[:, 1], 
                 color='red', alpha=0.2, label='95% Confidence Interval')
plt.title(f'ARIMA{best_order} - GDP Forecast with Confidence Intervals')
plt.xlabel('Time Period')
plt.ylabel('GDP')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()

# Calculate error metrics for the model
from sklearn.metrics import mean_absolute_error, mean_squared_error

# In-sample predictions
train_pred = final_results.predict(start=0, end=len(df)-1)
mae = mean_absolute_error(df["GDP"].iloc[1:], train_pred.iloc[1:])  # Skip first if differenced

mape = np.mean(np.abs((df["GDP"].iloc[1:] - train_pred.iloc[1:]) / df["GDP"].iloc[1:])) * 100

print(f"\n{'-'*50}")
print("Model Performance Metrics:")
print(f"{'-'*50}")
print(f"Mean Absolute Error (MAE): {mae:.2f}")
print(f"Root Mean Square Error (RMSE): {rmse:.2f}")
print(f"Mean Absolute Percentage Error (MAPE): {mape:.2f}%")
